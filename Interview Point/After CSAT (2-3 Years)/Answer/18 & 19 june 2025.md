 ```csharp
 public static void ReverseString_UsingArrayReverse(string input)
 {
     char[] arr = input.ToCharArray();
     Array.Reverse(arr);
     string reversed = new string(arr);
     Console.WriteLine("Reversed: " + reversed);
 }
 public static void ReverseString_UsingLoop(string input)
 {
     string reversed = "";
     for (int i = input.Length - 1; i >= 0; i--)
     {
         reversed += input[i];
     }
     Console.WriteLine(reversed);
 }
 public static void ReverseString_UsingLinq(string input)
 {
     string reversed = new string(input.Reverse().ToArray());
     Console.WriteLine(reversed);
 }
```

 ### find Dublicate String
 ```csharp
 public static void DublicateString_UsingLINQ(string [] arr)
 {
     var duplicates = arr
    .GroupBy(x => x)
    .Where(g => g.Count() > 1)
    .Select(g => g.Key);

     foreach (var item in duplicates)
     {
         Console.WriteLine(item);
     }
     
 }
 public static void DublicateString_UsingDictionary(string[] arr)
 {
     Dictionary<string, int> dict = new Dictionary<string, int>();

     foreach (var item in arr)
     {
         if (dict.ContainsKey(item))
             dict[item]++;
         else
             dict[item] = 1;
     }

     foreach (var pair in dict)
     {
         if (pair.Value > 1)
             Console.WriteLine(pair.Key);
     }
 }
 public static void DublicateString_UsingHashSet(string[] arr)
 {
     HashSet<string> seen = new HashSet<string>();
     HashSet<string> duplicates = new HashSet<string>();

     foreach (var item in arr)
     {
         if (!seen.Add(item))
         {
             duplicates.Add(item);
         }
     }

     foreach (var item in duplicates)
     {
         Console.WriteLine(item);
     }
 }
```

 ### Print Pyramid
 ```csharp
 public static void Full_Pyramid(int n)
 {
     //Centered

     for (int i = 1; i <= n; i++)
     {
         // spaces
         for (int j = 1; j <= n - i; j++)
         {
             Console.Write(" ");
         }

         // stars
         for (int k = 1; k <= (2 * i - 1); k++)
         {
             Console.Write("*");
         }

         Console.WriteLine();
     }
 }
 public static void Half_Pyramid(int n)
 {
     //Left Side
     for (int i = 1; i <= n; i++)
     {
         for (int j = 1; j <= i; j++)
         {
             Console.Write("*");
         }
         Console.WriteLine();
     }
 }
 public static void Inverted_Pyramid(int n)
 {
     for (int i = n; i >= 1; i--)
     {
         for (int j = 1; j <= i; j++)
         {
             Console.Write("*");
         }
         Console.WriteLine();
     }
 }

 public static void ReverseCentered_Pyramid(int n)
 {

     for (int i = n; i >= 1; i--)
     {
         for (int j = 1; j <= n - i; j++)
         {
             Console.Write(" ");
         }

         for (int k = 1; k <= (2 * i - 1); k++)
         {
             Console.Write("*");
         }

         Console.WriteLine();
     }
 }
 ```
